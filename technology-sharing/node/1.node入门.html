var fs = require('fs');
module.exports = {
    // 同步读取
    readFileSync: function (path, charset = 'utf-8') {
        var data = fs.readFileSync(path, charset);
        return data;
    },
    // 异步读取
    readFile: function (path, callback, charset = 'utf-8', ) {
        fs.readFile(path, function (err, data) {
            if (err) {
                callback('error')
                // console.log('读取文件出错');
            }else{
               callback(data);
            }
        })
    }

}

 var fs = require('fs');
module.exports = {
    readFile: function (path, callback) {
        fs.readFile(path, function (err, data) {
            if (err) {
                callback('error')
            } else {
                callback(data);
            }
        })
    },
    readImg: function (path, callback) {
        fs.readFile(path, 'binary',function (err,data) {
            if (err) {
                callback('error');
            } else {
                callback(data);
            }
        })

    }
}

var filesys = require('./filesys');
module.exports = {
    '/login.html': function (req, res) {
        filesys.readFile('login.html', function (e) {
            if (e == 'error') {
                res.writeHead(404, {
                    "Content-Type": "text/html;charset=utf-8"
                });
                res.write("文件读取出错");
                res.end();
            } else {
                res.write(e);
                res.end();
            }
        });
    },
    '/1.jpg': function (req, res) {
        filesys.readImg('1.jpg', function (e) {
            if (e == "error") {
                res.writeHead(404, {
                    "Content-Type": "Image/image"
                });
                res.write("文件读取出错");
                res.end();
            } else {
                res.write(e,'binary');
                res.end();
            }
        });
    }
}
