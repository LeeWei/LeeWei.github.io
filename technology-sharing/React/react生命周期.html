# 过滤器

// 过滤器-->文本数据化的处理
// react没有过滤器需要自己数据处理
    import React, { Component } from "react"
    class Hello extends React.Component{
        constructor(){
            super()
            this.state={
                data:0
            }
        };
        action(n){
            if(n>18){
                return "未成年"
            }else{
                return "已成年"
            }
        }
        action2(n){
           switch(n){
               case 1:
                   return "上线"
                case 2:
                   return "离线"
                default:
                    return n
           }
        }
        render(){
            const user={
                name:"abc",
                age:"18",
                n:2
            }
            return (
                <div>
                    {this.action(user.age)}
                    {this.action2(user.n)}
                </div>
            )
        }
    };
    ReactDOM.render(
        <Hello/>,
        document.getElementById("my")
    )
